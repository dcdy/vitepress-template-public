  1. 新建docs/.vitepress/theme/view-transition.ts文件

    view-transition.ts
    ts
    import { nextTick } from 'vue';
import { useData } from 'vitepress';

export function useAppearanceTransition() {
	const { isDark } = useData();

	// 判断是否支持 View Transition 和动画效果
	const enableTransitions = () =>
		'startViewTransition' in document && window.matchMedia('(prefers-reduced-motion: no-preference)').matches;

	// 提供 toggle-appearance 方法，切换主题
	const toggleAppearance = async ({ clientX: x, clientY: y }: MouseEvent) => {
		if (!enableTransitions()) {
			isDark.value = !isDark.value;
			return;
		}

		const clipPath = [
			`circle(0px at ${x}px ${y}px)`,
			`circle(${Math.hypot(Math.max(x, innerWidth - x), Math.max(y, innerHeight - y))}px at ${x}px ${y}px)`,
		];

		await document.startViewTransition(async () => {
			isDark.value = !isDark.value;
			await nextTick();
		}).ready;

		document.documentElement.animate(
			{ clipPath: isDark.value ? clipPath.reverse() : clipPath },
			{
				duration: 300,
				easing: 'ease-in',
				pseudoElement: `::view-transition-${isDark.value ? 'old' : 'new'}(root)`,
			}
		);
	};

	return {
		toggleAppearance,
	};
}

地址：https://github.com/catdad/canvas-confetti

shell
pnpm add canvas-confetti
1

配置 ​

  1. 新建组件.vitepress/theme/components/Confetti.vue

    点击查看Confetti.vue文件内容
    Confetti.vue
    vue
    <template></template>
<script setup lang="ts">
import confetti from 'canvas-confetti';
import { onMounted } from 'vue';

onMounted(() => {
	/* 大量纸屑效果 */
	largeConfetti();
});
/* 简易纸屑效果 */
const simpleConfetti = () => {
	confetti({
		particleCount: 100,
		spread: 170,
		origin: { y: 0.6 },
	});
};
/* 大量纸屑效果 */
const largeConfetti = () => {
	var count = 200;
	var defaults = {
		origin: { y: 0.7 },
	};

	function fire(particleRatio, opts) {
		confetti({
			...defaults,
			...opts,
			particleCount: Math.floor(count * particleRatio),
		});
	}

	fire(0.25, {
		spread: 26,
		startVelocity: 55,
	});
	fire(0.2, {
		spread: 60,
	});
	fire(0.35, {
		spread: 100,
		decay: 0.91,
		scalar: 0.8,
	});
	fire(0.1, {
		spread: 120,
		startVelocity: 25,
		decay: 0.92,
		scalar: 1.2,
	});
	fire(0.1, {
		spread: 120,
		startVelocity: 45,
	});
};
/* 两侧纸屑效果 */
const sideConfetti = () => {
	var end = Date.now() + 3 * 1000;
	var colors = ['#FC7792', '#FCB34F'];

	(function frame() {
		confetti({
			particleCount: 2,
			angle: 60,
			spread: 55,
			origin: { x: 0 },
			colors: colors,
		});
		confetti({
			particleCount: 2,
			angle: 120,
			spread: 55,
			origin: { x: 1 },
			colors: colors,
		});

		if (Date.now() < end) {
			requestAnimationFrame(frame);
		}
	})();
};
</script>

  2. 在 index.ts 中注册全局组件

    index.ts
    ts
    // 五彩纸屑
import Confetti from './components/Confetti.vue';

export default {
	extends: DefaultTheme,
	enhanceApp({ app }) {
		// 全局注册五彩纸屑组件
		app.component('Confetti', Confetti);
	},
};

<!-- 五彩纸屑 -->
